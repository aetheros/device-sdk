279 m2m::ResponsePrimitive_ptr createSimpleSubscription( std::string const& target, std::string const& name,
280 xsd::m2m::EventNotificationCriteria eventNotificationCriteria )
281 {
282 auto subscription = xsd::m2m::Subscription::Create();
283 subscription.creator = std::string();
284
285 // set the subscription's name

287
288 // have all resource attributes be provided in the notifications

297
298 auto request = newRequest( xsd::m2m::Operation::Create, m2m::To{ target } );
299 request.req->resourceType = xsd::m2m::ResourceType::subscription;
300 request.req->resultContent = xsd::m2m::ResultContent::Nothing;
301 request.req->primitiveContent = xsd::toAnyNamed( subscription );
302
303 sendRoleRequest( request );
304 return getResponse( request );
