38/*! \brief struct struct sent from the Waveform Provider service containing stream and sample information.
39 * As the service detects errors, sync will be incremented. sequence will increment normally, but be reset
40 * when sync changes. So, under normal successfull operation, consecutive packets will have the same sync
41 * value, but increasing sequence values. If sync changes, there was a capture error and samples were lost.
42 * If sync does not change, but sequence increased by more than one, then at least one packet was lost
43 * between the service and the application.
44 */
45struct Packet
46{
47 Network network; //!< The /ref Network type sampled to generate the /ref Packet.
48 uint16_t sync; //!< Incremented every time the service detects an invalid crc or fails to read the capture device.
49 uint16_t sequence; //!< Incremented every packet. Reset to 0 when sync changes.
50 SampleBlock blocks[64]; //!< 64 consecutive /ref SampleBlock read from the capture device.
51};
